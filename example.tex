\begin{lstlisting}[language=GAPsrc]
Bahn := function(erz, a, wirk)
  local delta, b, g, c;
  delta := [a];
  for b in delta do
      for g in erz do
          c := wirk(b, g);
          if not c in delta then
              Add(delta, c);
          fi;
      od;
  od;
  return delta;
end;
\end{lstlisting}
